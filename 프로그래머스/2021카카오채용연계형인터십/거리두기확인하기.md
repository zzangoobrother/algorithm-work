import java.util.*;

public class algorithm {
  static int[] dx = {1, 0, -1, 0};
	static int[] dy = {0, -1, 0, 1};
  
  public static void main(String[] args) {
    String[][] places = {{"POOOP", "OXXOX", "OPXPX", "OOXOX", "POXXP"}, 
				{"POOPX", "OXPXP", "PXXXO", "OXXXO", "OOOPP"}, 
				{"PXOPX", "OXOXP", "OXPOX", "OXXOP", "PXPOX"}, 
				{"OOOXX", "XOOOX", "OOOXX", "OXOOX", "OOOOO"}, 
				{"PXPXP", "XPXPX", "PXPXP", "XPXPX", "PXPXP"}};
		
		int[] answer = new int[places.length];
		for (int i = 0; i < 5; i++) {
			if (check(places[i])) {
				answer[i] = 1;
			} else {
				answer[i] = 0;
			}
		}
		
		
		for (int i : answer) {
			System.out.println(i);
		}
  }
  
  static boolean check(String[] place) {
		for (int i = 0; i < 5; i++){
			for (int j = 0; j < 5; j++) {
				if (place[i].charAt(j) == 'P') {
					if (bfs(place, i, j) == false)
						return false;
				}
			}
		}
		return true;
	}
  
  static boolean bfs(String[] place, int x, int y) {
		boolean[][] visit = new boolean[5][5];
		
		Queue<Point> queue = new LinkedList<>();
		visit[x][y] = true;
		queue.offer(new Point(x, y, 0));
		while (!queue.isEmpty()) {
			Point point = queue.poll();
			
			if (point.dis > 2) 
				continue;
			
			if (point.dis != 0 && place[point.x].charAt(point.y) == 'P') {
				return false;
			}
			
			for (int i = 0; i < 4; i++) {
				int nx = point.x + dx[i];
				int ny = point.y + dy[i];
				if (nx < 0 || nx > 4 || ny < 0 || ny > 4)
					continue;
				
				if (visit[nx][ny])
					continue;
				
				if (place[nx].charAt(ny) == 'X')
					continue;
				
				visit[nx][ny] = true;
				queue.offer(new Point(nx, ny, point.dis + 1));
			}
		}
		
		return true;
	}
}

class Point {
	int x;
	int y;
	int dis;
	Point(int x, int y, int dis) {
		this.x = x;
		this.y = y;
		this.dis = dis;
	}
}
