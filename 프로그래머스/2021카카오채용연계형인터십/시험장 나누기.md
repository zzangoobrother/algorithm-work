import java.util.*;

public class algorithm {
  static int[] left;
	static int[] right;
	static int[] parent;
	static int[] people;
	static int root;
	static int cnt;
  
  public static void main(String[] args) {
    int k = 3;
		int[] num = {12, 30, 1, 8, 8, 6, 20, 7, 5, 10, 4, 1};
		int[][] links = {{-1, -1}, {-1, -1}, {-1, -1}, {-1, -1}, {8, 5},
				{2, 10}, {3, 0}, {6, 1}, {11, -1}, {7, 4}, {-1, -1}, {-1, -1}};
		
		left = new int[num.length];
		right = new int[num.length];
		parent = new int[num.length];
		people = num;
    
    Arrays.fill(parent, -1);
		
		for (int i = 0; i < links.length; i++) {
			left[i] = links[i][0];
			right[i] = links[i][1];
			
			if (left[i] != -1) {
				parent[left[i]] = i;
			}
			if (right[i] != -1) {
				parent[right[i]] = i;
			}
		}
		
		for (int i = 0; i < num.length; i++) {
			if (parent[i] == -1) {
				root = i;
				break;
			}
		}
		
		int start = 0;
		int end =(int) 1e9; 
		for (int i = 0; i < num.length; i++) {
			start = Math.max(start, people[i]);
		}
		
		while (start <= end) {
			int mid = (start + end) / 2;
			if (getGroup(mid) <= k) {
				end = mid - 1;
			} else {
				start = mid + 1;
			}
		}
		
		System.out.println(end+1);
  }
  
  static int getGroup(int limit) {
		cnt = 0;
		dfs(root, limit);
		return cnt + 1;
	}
	
	static int dfs(int root, int limit) {
		int lv = 0;
		int rv = 0;
		
		if (left[root] != -1) {
			lv = dfs(left[root], limit);
		}
		
		if (right[root] != -1) {
			rv = dfs(right[root], limit);
		}
		
		if (people[root] + lv + rv <= limit) {
			return people[root] + lv + rv;
		}
		
		if (people[root] + Math.min(lv, rv) <= limit) {
			cnt++;
			return people[root] + Math.min(lv, rv);
		}
		
		cnt += 2;
		return people[root];
	}
}
