# 크레인 인형뽑기 게임

출처 : https://programmers.co.kr/learn/courses/30/lessons/64061

- java
```java
import java.util.*

public class algorithm {
  public static void main(String[] args) {
    Scanner in=new Scanner(System.in);
	    int input1 = in.nextInt();
		int[][] board = new int[input1][input1];
		for (int i = 0; i < input1; i++) {
			for (int j = 0; j < input1; j++) {
				board[i][j] = in.nextInt();
			}
		}
		
		int input2 = in.nextInt();
		int[] moves = new int[input2];
		for (int i = 0; i < input2; i++) {
			moves[i] = in.nextInt();
		}
    
    Stack<Integer> stack = new Stack<>();
    int result = 0;
		for (int x : moves) {
			for (int i = 0; i < input1; i++) {
				if (board[i][x - 1] != 0) {
					int temp = board[i][x - 1];
					board[i][x - 1] = 0;
					if (!stack.isEmpty() && temp == stack.peek()) {
						stack.pop();
						result += 2;
					} else {
						stack.push(temp);
					}
					break;
				}
			}
		}
  }
}
```

input1 = int(input())
board = []
for i in range(input1):
    board.append(input().strip().split(' '))

input2 = int(input())
moves = input().strip().split(' ')

stack = []
result = 0
for x in moves:
    for i in range(input1):
        if int(board[i][int(x) - 1]) != 0:
            temp = int(board[i][int(x) - 1])
            board[i][int(x) - 1] = 0
            if stack != [] and stack[-1] == temp:
                stack.pop()
                result += 2
            else:
                stack.append(temp)
            break

print(result)
